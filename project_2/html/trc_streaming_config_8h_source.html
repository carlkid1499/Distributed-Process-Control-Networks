53 /*******************************************************************************
54  * Configuration Macro: TRC_CFG_SYMBOL_TABLE_SLOTS
55  *
56  * The maximum number of symbols names that can be stored. This includes:
57  * - Task names
58  * - Named ISRs (vTraceSetISRProperties)
59  * - Named kernel objects (vTraceStoreKernelObjectName)
60  * - User event channels (xTraceRegisterString)
61  *
62  * If this value is too small, not all symbol names will be stored and the
63  * trace display will be affected. In that case, there will be warnings
64  * (as User Events) from TzCtrl task, that monitors this.
65  ******************************************************************************/
66 #define TRC_CFG_SYMBOL_TABLE_SLOTS 40
67 
68 /*******************************************************************************
69  * Configuration Macro: TRC_CFG_SYMBOL_MAX_LENGTH
70  *
71  * The maximum length of symbol names, including:
72  * - Task names
73  * - Named ISRs (vTraceSetISRProperties)
74  * - Named kernel objects (vTraceStoreKernelObjectName)
75  * - User event channel names (xTraceRegisterString)
76  *
77  * If longer symbol names are used, they will be truncated by the recorder,
78  * which will affect the trace display. In that case, there will be warnings
79  * (as User Events) from TzCtrl task, that monitors this.
80  ******************************************************************************/
81 #define TRC_CFG_SYMBOL_MAX_LENGTH 25
82 
83 /*******************************************************************************
84  * Configuration Macro: TRC_CFG_OBJECT_DATA_SLOTS
85  *
86  * The maximum number of object data entries (used for task priorities) that can
87  * be stored at the same time. Must be sufficient for all tasks, otherwise there
88  * will be warnings (as User Events) from TzCtrl task, that monitors this.
89  ******************************************************************************/
90 #define TRC_CFG_OBJECT_DATA_SLOTS 40
91 
92 /*******************************************************************************
93  * Configuration Macro: TRC_CFG_PAGED_EVENT_BUFFER_PAGE_COUNT
94  *
95  * Specifies the number of pages used by the paged event buffer.
96  * This may need to be increased if there are a lot of missed events.
97  *
98  * Note: not used by the J-Link RTT stream port (see trcStreamingPort.h instead)
99  ******************************************************************************/
100 #define TRC_CFG_PAGED_EVENT_BUFFER_PAGE_COUNT 10
101 
102 /*******************************************************************************
103  * Configuration Macro: TRC_CFG_PAGED_EVENT_BUFFER_PAGE_SIZE
104  *
105  * Specifies the size of each page in the paged event buffer. This can be tuned
106  * to match any internal low-level buffers used by the streaming interface, like
107  * the Ethernet MTU (Maximum Transmission Unit). However, since the currently
108  * active page can't be transfered, having more but smaller pages is more
109  * efficient with respect memory usage, than having a few large pages.
110  *
111  * Note: not used by the J-Link RTT stream port (see trcStreamingPort.h instead)
112  ******************************************************************************/
113 #define TRC_CFG_PAGED_EVENT_BUFFER_PAGE_SIZE 500
114 
115 /*******************************************************************************
116  * TRC_CFG_ISR_TAILCHAINING_THRESHOLD
117  *
118  * Macro which should be defined as an integer value.
119  *
120  * If tracing multiple ISRs, this setting allows for accurate display of the
121  * context-switching also in cases when the ISRs execute in direct sequence.
122  *
123  * vTraceStoreISREnd normally assumes that the ISR returns to the previous
124  * context, i.e., a task or a preempted ISR. But if another traced ISR
125  * executes in direct sequence, Tracealyzer may incorrectly display a minimal
126  * fragment of the previous context in between the ISRs.
127  *
128  * By using TRC_CFG_ISR_TAILCHAINING_THRESHOLD you can avoid this. This is
129  * however a threshold value that must be measured for your specific setup.
130  * See http://percepio.com/2014/03/21/isr_tailchaining_threshold/
131  *
132  * The default setting is 0, meaning "disabled" and that you may get an
133  * extra fragments of the previous context in between tail-chained ISRs.
134  *
135  * Note: This setting has separate definitions in trcSnapshotConfig.h and
136  * trcStreamingConfig.h, since it is affected by the recorder mode.
137  ******************************************************************************/
138 #define TRC_CFG_ISR_TAILCHAINING_THRESHOLD 0
