241 /*******************************************************************************
242  * uiTraceStart
243  *
244  * Starts the recorder. The recorder will not be started if an error has been
245  * indicated using prvTraceError, e.g. if any of the Nx constants in trcConfig.h
246  * has a too small value (TRC_CFG_NTASK, TRC_CFG_NQUEUE, etc).
247  *
248  * Returns 1 if the recorder was started successfully.
249  * Returns 0 if the recorder start was prevented due to a previous internal
250  * error. In that case, check xTraceGetLastError to get the error message.
251  * Any error message is also presented when opening a trace file.
252  *
253  * This function is obsolete, but has been saved for backwards compatibility.
254  * We recommend using vTraceEnable instead.
255  ******************************************************************************/
256 uint32_t uiTraceStart(void)
257 {
258  traceHandle handle;
259  TRACE_ALLOC_CRITICAL_SECTION();
260 
261  handle = 0;
262 
263  if (RecorderDataPtr == NULL)
264  {
265  TRACE_ASSERT(RecorderDataPtr != NULL, "Recorder not initialized. Use vTraceEnable() instead!", 0);
266  return 0;
267  }
268 
269  if (RecorderDataPtr->recorderActive == 1)
270  return 1; /* Already running */
271 
272  if (traceErrorMessage == NULL)
273  {
274  trcCRITICAL_SECTION_BEGIN();
275  RecorderDataPtr->recorderActive = 1;
276 
277  handle = TRACE_GET_TASK_NUMBER(TRACE_GET_CURRENT_TASK());
278  if (handle == 0)
279  {
280  /* This occurs if the scheduler is not yet started.
281  This creates a dummy "(startup)" task entry internally in the
282  recorder */
283  handle = prvTraceGetObjectHandle(TRACE_CLASS_TASK);
284  prvTraceSetObjectName(TRACE_CLASS_TASK, handle, "(startup)");
285 
286  prvTraceSetPriorityProperty(TRACE_CLASS_TASK, handle, 0);
287  }
288 
289  prvTraceStoreTaskswitch(handle); /* Register the currently running task */
290  trcCRITICAL_SECTION_END();
291  }
292 
293  return RecorderDataPtr->recorderActive;
294 }

525 /*******************************************************************************
526  * vTraceStoreISRBegin
527  *
528  * Registers the beginning of an Interrupt Service Routine, using a traceHandle
529  * provided by xTraceSetISRProperties.
530  *
531  * Example:
532  * #define PRIO_ISR_TIMER1 3 // the hardware priority of the interrupt
533  * ...
534  * traceHandle Timer1Handle = xTraceSetISRProperties("ISRTimer1", PRIO_ISR_TIMER1);
535  * ...
536  * void ISR_handler()
537  * {
538  * vTraceStoreISRBegin(Timer1Handle);
539  * ...
540  * vTraceStoreISREnd(0);
541  * }
542  ******************************************************************************/
543 void vTraceStoreISRBegin(traceHandle handle)
544 {
545  TRACE_ALLOC_CRITICAL_SECTION();
546 
547  if (recorder_busy)
548  {
549  /*************************************************************************
550  * This occurs if an ISR calls a trace function, preempting a previous
551  * trace call that is being processed in a different ISR or task.
552  * If this occurs, there is probably a problem in the definition of the
553  * recorder's internal critical sections (TRACE_ENTER_CRITICAL_SECTION and
554  * TRACE_EXIT_CRITICAL_SECTION). They must disable the RTOS tick interrupt
555  * and any other ISRs that calls the trace recorder directly or via
556  * traced kernel functions. The ARM port disables all interrupts using the
557  * PRIMASK register to avoid this issue.
558  *************************************************************************/
559  prvTraceError("vTraceStoreISRBegin - recorder busy! See code comment.");
560  return;
561  }
562  trcCRITICAL_SECTION_BEGIN();
563 
564  if (RecorderDataPtr->recorderActive && handle_of_last_logged_task)
565  {
566  uint16_t dts4;
567 
568  TRACE_ASSERT(handle != 0, "vTraceStoreISRBegin: Invalid ISR handle (NULL)", TRC_UNUSED);
569  TRACE_ASSERT(handle <= RecorderDataPtr->ObjectPropertyTable.NumberOfObjectsPerClass[TRACE_CLASS_ISR], "vTraceStoreISRBegin: Invalid ISR handle (> NISR)", TRC_UNUSED);
570 
571  dts4 = (uint16_t)prvTraceGetDTS(0xFFFF);
572 
573  if (RecorderDataPtr->recorderActive) /* Need to repeat this check! */
574  {
575  if (nISRactive < TRC_CFG_MAX_ISR_NESTING)
576  {
577  TSEvent* ts;
578  uint8_t hnd8 = prvTraceGet8BitHandle(handle);
579  isrstack[nISRactive] = handle;
580  nISRactive++;
581  ts = (TSEvent*)prvTraceNextFreeEventBufferSlot();
582  if (ts != NULL)
583  {
584  ts->type = TS_ISR_BEGIN;
585  ts->dts = dts4;
586  ts->objHandle = hnd8;
587  prvTraceUpdateCounters();
588  }
589  }
590  else
591  {
592  /* This should not occur unless something is very wrong */
593  prvTraceError("Too many nested interrupts!");
594  }
595  }
596  }
597  trcCRITICAL_SECTION_END();
598 }

1237 /******************************************************************************
1238  * vTracePrintF
1239  *
1240  * Generates User Event with formatted text and data, similar to a "printf".
1241  * It is very fast compared to a normal "printf" since this function only
1242  * stores the arguments. The actual formatting is done
1243  * on the host PC when the trace is displayed in the viewer tool.
1244  *
1245  * User Event labels are created using xTraceRegisterString.
1246  * Example:
1247  *
1248  * traceString adc_uechannel = xTraceRegisterString("ADC User Events");
1249  * ...
1250  * vTracePrintF(adc_uechannel,
1251  * "ADC channel %d: %lf volts",
1252  * ch, (double)adc_reading/(double)scale);
1253  *
1254  * This can be combined into one line, if desired, but this is slower:
1255  *
1256  * vTracePrintF(xTraceRegisterString("ADC User Events"),
1257  * "ADC channel %d: %lf volts",
1258  * ch, (double)adc_reading/(double)scale);
1259  *
1260  * Calling xTraceRegisterString multiple times will not create duplicate entries, but
1261  * it is of course faster to just do it once, and then keep the handle for later
1262  * use. If you don't have any data arguments, only a text label/string, it is
1263  * better to use vTracePrint - it is faster.
1264  *
1265  * Format specifiers supported:
1266  * %d - 32 bit signed integer
1267  * %u - 32 bit unsigned integer
1268  * %f - 32 bit float
1269  * %s - string (is copied to the recorder symbol table)
1270  * %hd - 16 bit signed integer
1271  * %hu - 16 bit unsigned integer
1272  * %bd - 8 bit signed integer
1273  * %bu - 8 bit unsigned integer

1282  * a double (%lf) uses two records, so this is quite costly. Use float (%f)
1283  * unless the higher precision is really necessary.
1284  *
1285  * Note that the double-precision float (%lf) assumes a 64 bit double
1286  * representation. This does not seem to be the case on e.g. PIC24 and PIC32.
1287  * Before using a %lf argument on a 16-bit MCU, please verify that
1288  * "sizeof(double)" actually gives 8 as expected. If not, use %f instead.
