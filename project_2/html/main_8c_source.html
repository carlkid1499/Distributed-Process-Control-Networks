168 /* vLEDC_ISR_Handler Function Description ****************************************

171  * DESCRIPTION: ISR Function called from ISR Assembly Wrapper

174  * NOTES: ISR C Code for Assembly Wrapper
175  * END DESCRIPTION ************************************************************/
176 void vLEDC_ISR_Handler(void) {
177  mCNIntEnable(0); // disable CN interrupts at beginning
178  #if ( HOME_PRO_MX7_BOARD == 1 )
179  LATGSET = LED4; // set the LED4 since we entered the ISR
180  #else
181  LATBSET = LEDD; // set the LEDD since we entered the ISR
182  #endif
183  #if ( configUSE_TRACE_FACILITY == 1 )
184  vTracePrint(str, "LEDD on from ISR");
185  #endif
186  int SValue = 1;
187  hw_msDelay(20); // 20 ms button debounce
188  while (PORTReadBits(IOPORT_G, BTN1)); // poll button 1 on
189  portBASE_TYPE xHigherPriorityTaskWoken = NULL;
190  portBASE_TYPE QStatus = xQueueSendToBackFromISR(QBTN1Status, &SValue, &xHigherPriorityTaskWoken); // Send BTN1 status to queue
191  hw_msDelay(20); // 20 ms button debounce
192  xHigherPriorityTaskWoken = pdFALSE;
193  /* Let's give a semaphore to unblock LEDCHandler_Task*/
194  xSemaphoreGiveFromISR(LEDC_Semaphore, &xHigherPriorityTaskWoken);
195 
196  mCNClearIntFlag(); // Macro function to clear CNI flag
197 
198  /* Giving the semaphore may have unblocked a task -if it did and the
199  * unblocked task has a priority equal to or above the currently executing
200  * task then xHigherPriorityTaskWoken will have been set to pdTRUE
201  * and portEND_SWITCHING_ISR() will force a context switch to the newly
202  * unblocked higher priority task.NOTE: The syntax for forcing a context
203  * switch within an ISR varies between FreeRTOS ports. The
204  * portEND_SWITCHING_ISR() macro is provided as part of
205  * the PIC32 port layer for this purpose. taskYIELD() must never be
206  * called from an ISR! */
207  #if ( HOME_PRO_MX7_BOARD == 1 )
208  LATGCLR = LED4; // unset the LED4 since we are exiting the ISR
209  #else
210  LATBCLR = LEDD; // unset the LEDD since we are exiting the ISR
211  #endif
212  #if ( configUSE_TRACE_FACILITY == 1 )
213  vTracePrint(str, "LEDD off from ISR");
214  #endif
215  mCNIntEnable(1); // Enable CN interrupts at end
216  portEND_SWITCHING_ISR(xHigherPriorityTaskWoken);
217 } /* End of vLEDC_ISR_Handler */

271 void vApplicationIdleHook(void) {
272  /* vApplicationIdleHook() will only be called if configUSE_IDLE_HOOK is set
273  to 1 in FreeRTOSConfig.h. It will be called on each iteration of the idle
274  task. It is essential that code added to this hook function never attempts
275  to block in any way (for example, call xQueueReceive() with a block time
276  specified, or call vTaskDelay()). If the application makes use of the
277  vTaskDelete() API function (as this demo application does) then it is also
278  important that vApplicationIdleHook() is permitted to return to its calling
279  function, because it is the responsibility of the idle task to clean up
280  memory allocated by the kernel to any task that has since been deleted. */
281 
282  int RValue = 0;
